package model;

public class Salario {

    private float salarioBruto;
    private float descontos;
    private float extras;
    private float salarioLiquido;

    public float getSalarioBruto() {
        return salarioBruto;
    }

    public void setSalarioBruto(float salarioBruto) {
        this.salarioBruto = salarioBruto;
    }

    public float getDescontos() {
        return descontos;
    }

    public float getExtras() {
        return extras;
    }

    public float getSalarioLiquido() {
        return salarioLiquido;
    }

    public Salario(int idFuncionario ,float salarioBruto, float descontos, float extras, float salarioLiquido) {
        this.salarioBruto = salarioBruto;
        this.descontos = descontos;
        this.extras = extras;
        this.salarioLiquido = salarioBruto + extras - descontos;
    }

    @Override
    public String toString() {
        return  "   | Salário Bruto : " + salarioBruto +
                "   | Descontos : " + descontos +
                "   | Extras : " + extras +
                "   | Salário Liquido : " + salarioLiquido ;
    }


}
